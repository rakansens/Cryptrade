function ema(values: number[], period: number): number[] {
  const k = 2 / (period + 1)
  const emaArr: number[] = []
  values.forEach((v, i) => {
    if (i === 0) {
      emaArr.push(v)
    } else {
      emaArr.push(v * k + emaArr[i - 1] * (1 - k))
    }
  })
  return emaArr
}
